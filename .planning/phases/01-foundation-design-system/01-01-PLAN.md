    ```css
    /* Font integration — MUST use @theme inline for CSS variable references (per RESEARCH.md Pitfall 1) */
    @theme inline {
      --font-sans: var(--font-geist-sans);
      --font-mono: var(--font-geist-mono);
    }

    @theme {
      /* Colors - dark theme (per D-01: dark charcoal background, D-02: electric blue accent) */
      --color-background: #0a0a0a;

      --color-muted: #a1a1aa;

      --color-border: #27272a;

      --color-accent: #3b82f6;
      --color-accent-hover: #60a5fa;
      --color-accent-muted: #3b82f615;

      /* Typography scale — 4 sizes per UI-SPEC */
      --text-hero: 3rem;
      --text-heading: 1.5rem;
      --text-body: 1rem;
      --text-sm: 0.875rem;

      /* Line heights per UI-SPEC */
      --leading-hero: 1.1;
      --leading-heading: 1.2;
      --leading-body: 1.6;

      /* Spacing base unit — Tailwind v4 derives all utilities from this (p-1=4px, p-2=8px, etc.) */
      --spacing: 0.25rem;
    }

    /* Background gradient for visual depth (per D-04, DSGN-05) */
    body {
      background-color: var(--color-background);
      background-image: radial-gradient(
        ellipse 80% 50% at 50% -20%,
        rgba(59, 130, 246, 0.08),
        transparent
      );
      background-attachment: fixed;
    }
    ```

    **CRITICAL RULES:**
    - The `@theme inline` block MUST be separate from the `@theme` block. Font CSS variables that reference `var(--font-geist-sans)` MUST use `@theme inline` — without `inline`, fonts silently fail (RESEARCH.md Pitfall 1).
    - Do NOT create a `tailwind.config.js` or `tailwind.config.ts` file. Tailwind v4 uses CSS-first config only.
    - The gradient opacity MUST be 0.08, not higher. The RGB values (59, 130, 246) match --color-accent (#3b82f6).
    - `background-attachment: fixed` ensures the gradient does not scroll with content.

    ```tsx
    import type { Metadata } from "next";
    import { Geist, Geist_Mono } from "next/font/google";
    import "./globals.css";

    const geistSans = Geist({
      variable: "--font-geist-sans",
      subsets: ["latin"],
    });

    const geistMono = Geist_Mono({
      variable: "--font-geist-mono",
      subsets: ["latin"],
    });

    export const metadata: Metadata = {
      title: "Moksh Verma | Backend Engineer",
      description:
        "Backend Engineer with 5+ years of experience building distributed systems at scale.",
    };

    export default function RootLayout({
      children,
    }: {
      children: React.ReactNode;
    }) {
      return (
        <html lang="en" className={`${geistSans.variable} ${geistMono.variable}`}>
          <body className="bg-background text-foreground font-sans antialiased">
            <header role="banner">
              {/* Navigation — built in Phase 3 */}
            </header>
            <main role="main">{children}</main>
            <footer role="contentinfo">
              {/* Footer — built in Phase 2 */}
            </footer>
          </body>
        </html>
      );
    }
    ```

    **CRITICAL RULES:**
    - Font loading uses the `variable` option (NOT `className`). This sets CSS custom properties `--font-geist-sans` and `--font-geist-mono` on the `<html>` element.
    - Both variable classes MUST be applied to `<html>` (not `<body>`) so they cascade to all elements.
    - The `@theme inline` block in globals.css maps `--font-sans` to `var(--font-geist-sans)`, enabling the `font-sans` utility class.
    - `<html lang="en">` is required for accessibility (DSGN-04).
    - `<header role="banner">`, `<main role="main">`, `<footer role="contentinfo">` provide ARIA landmarks (DSGN-04).
    - The `bg-background text-foreground font-sans antialiased` classes apply the design tokens from globals.css.
    - Metadata sets "Moksh Verma | Backend Engineer" as page title per UI-SPEC copywriting contract.
